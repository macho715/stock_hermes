```python
# src/stock_rtx4060/advisors/langgraph_pipeline.py

from langgraph.graph import END, StateGraph
from typing import TypedDict

class AdvisorState(TypedDict):
    ticker: str
    factors: dict
    bull_score: float | None
    bear_score: float | None
    devils_score: float | None
    final_score: float | None

def build_advisor_graph() -> StateGraph:
    g = StateGraph(AdvisorState)
    g.add_node("bull", bull_agent)
    g.add_node("bear", bear_agent)
    g.add_node("devils", devils_advocate_agent)
    g.add_node("synthesize", synthesize_scores)
    # 병렬 실행: bull + bear + devils → synthesize
    g.add_edge("__start__", "bull")
    g.add_edge("__start__", "bear")
    g.add_edge("__start__", "devils")
    g.add_edge(["bull", "bear", "devils"], "synthesize")
    g.add_edge("synthesize", END)
    return g.compile()
```

```python
from fastapi import FastAPI, WebSocket
from fastapi.middleware.cors import CORSMiddleware
import asyncio, json

app = FastAPI()
app.add_middleware(CORSMiddleware, allow_origins=["http://localhost:5173", ...])

# 기존 Flask 엔드포인트 → FastAPI 로 1:1 마이그레이션
@app.get("/api/recommend")
async def recommend(tickers: str = "AAPL,MSFT"):
    ...  # 기존 로직 async wrapping

# 신규 WebSocket 엔드포인트
@app.websocket("/ws/quote/{ticker}")
async def ws_quote(websocket: WebSocket, ticker: str):
    await websocket.accept()
    try:
        while True:
            price = await get_latest_price(ticker)  # Alpaca or yfinance
            await websocket.send_json({"ticker": ticker, "price": price, "ts": ...})
            await asyncio.sleep(1)
    except Exception:
        await websocket.close()
```
